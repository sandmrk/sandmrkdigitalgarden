(function() {
    // 状态配置
    const CONFIG = {
        pageSize: 20,
        rootMargin: '600px',
        heartbeatInterval: 400,
        maxRetries: 25 // 约 10 秒后强制停止检测，节省电量
    };

    let state = {
        visibleCount: CONFIG.pageSize,
        isApplied: false,
        retryCount: 0
    };

    /**
     * 核心：执行无限滚动逻辑
     */
    function applyInfiniteScroll() {
        const container = document.getElementById('gallery-grid');
        const sentinel = document.getElementById('infinite-sentinel');
        if (!container || !sentinel) return false;

        const rows = container.querySelectorAll('table tbody tr');
        if (rows.length === 0) return false;

        // 更新可见性的函数
        const updateRows = () => {
            rows.forEach((row, i) => {
                row.style.setProperty('display', i < state.visibleCount ? 'flex' : 'none', 'important');
            });
            if (state.visibleCount >= rows.length) {
                sentinel.innerText = "已加载全部作品 💚";
            }
        };

        // 初始化 IntersectionObserver
        if (!state.isApplied) {
            const observer = new IntersectionObserver((entries) => {
                if (entries[0].isIntersecting && state.visibleCount < rows.length) {
                    state.visibleCount += CONFIG.pageSize;
                    updateRows();
                }
            }, { rootMargin: CONFIG.rootMargin, threshold: 0.1 });

            observer.observe(sentinel);
            state.isApplied = true;
        }

        updateRows();
        return true; // 标记成功执行
    }

    /**
     * 智能心跳检测
     * 解决 Vercel 静态站 Dataview 渲染延迟问题
     */
    const heartbeat = setInterval(() => {
        state.retryCount++;
        const success = applyInfiniteScroll();
        
        // 停止条件：1. 成功接管且已隐藏多余行 2. 超过最大尝试次数
        const rows = document.querySelectorAll('#gallery-grid table tbody tr');
        if ((success && rows.length > CONFIG.pageSize && rows[CONFIG.pageSize].style.display === 'none') || 
            state.retryCount > CONFIG.maxRetries) {
            clearInterval(heartbeat);
        }
    }, CONFIG.heartbeatInterval);

    /**
     * 处理 SPA 路由跳转
     * Digital Garden 在切换页面时不刷新浏览器，需重置状态
     */
    const resetState = () => {
        state.isApplied = false;
        state.visibleCount = CONFIG.pageSize;
        // 重新启动检测，适配从内容页返回首页的情况
        // 这里的逻辑可根据实际跳转表现微调
    };

    window.addEventListener('popstate', resetState);
})();
