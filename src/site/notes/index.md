<script>
document.addEventListener('DOMContentLoaded', () => {
  const initGallery = () => {
      let visibleCount = 20;
      let isApplied = false;

      const runEngine = () => {
          // 改用 class 选择器，因为 ID 容易被插件过滤
          const container = document.querySelector('.grid-view');
          const sentinel = document.querySelector('.infinite-sentinel-marker');
          const rows = document.querySelectorAll('.grid-view table tbody tr');

          if (!container || !sentinel || !rows.length) return;

          const update = () => {
              rows.forEach((row, i) => {
                  // 这里的 flex 必须配合你的 CSS，如果是普通表格请用 table-row
                  row.style.display = i < visibleCount ? '' : 'none';
              });
              if (visibleCount >= rows.length) sentinel.innerText = "已加载全部作品 💚";
          };

          if (!isApplied) {
              const observer = new IntersectionObserver((entries) => {
                  if (entries[0].isIntersecting && visibleCount < rows.length) {
                      visibleCount += 20;
                      update();
                  }
              }, { rootMargin: '400px' });
              observer.observe(sentinel);
              isApplied = true;
          }
          update(); 
      };

      const heartBeat = setInterval(() => {
          runEngine();
          const rows = document.querySelectorAll('.grid-view table tbody tr');
          if (isApplied && rows.length > 0) {
              if (rows.length <= 20 || (rows[20] && rows[20].style.display === 'none')) {
                  clearInterval(heartBeat);
              }
          }
      }, 500);
      setTimeout(() => clearInterval(heartBeat), 10000); 
  };
  
  initGallery();
});
</script>
